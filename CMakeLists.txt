cmake_minimum_required (VERSION 3.14)
project (CheckTypeSizeIOSTest)

include (CheckTypeSize)
check_type_size ("float" SIZEOF_FLOAT)
check_type_size ("double" SIZEOF_DOUBLE)
check_type_size ("int" SIZEOF_INT)
check_type_size ("long" SIZEOF_LONG)
check_type_size ("long long" SIZEOF_LONG_LONG)
check_type_size ("unsigned int" SIZEOF_UNSIGNED_INT)
check_type_size ("unsigned long" SIZEOF_UNSIGNED_LONG)
check_type_size ("unsigned long long" SIZEOF_UNSIGNED_LONG_LONG)
check_type_size ("size_t" SIZEOF_SIZE_T)
check_type_size ("ptrdiff_t" SIZEOF_PTRDIFF_T)

message("Sizeof float : ${SIZEOF_FLOAT}")
message("Sizeof double : ${SIZEOF_DOUBLE}")
message("Sizeof int : ${SIZEOF_INT}")
message("Sizeof long : ${SIZEOF_LONG}")
message("Sizeof long long : ${SIZEOF_LONG_LONG}")
message("Sizeof unsigned int : ${SIZEOF_UNSIGNED_INT}")
message("Sizeof unsigned long : ${SIZEOF_UNSIGNED_LONG}")
message("Sizeof unsigned long long : ${SIZEOF_UNSIGNED_LONG_LONG}")
message("Sizeof size_t : ${SIZEOF_SIZE_T}")
message("Sizeof ptrdiff_t : ${SIZEOF_PTRDIFF_T}")
